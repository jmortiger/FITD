#ifndef _DEBUG_H_
#define _DEBUG_H_
// #include "common.h"

/// @brief The categories of debug info supported, as a bit flag.
enum debugOutputEnum
{
	DBO_NONE	 = 0b00000000,
	DBO_PAK		 = 0b00000001,
	DBO_FLOOR	 = 0b00000010,
	DBO_MASK	 = 0b00000100,
	DBO_CAMERA	 = 0b00001000,
	DBO_SOUND	 = 0b00010000,
};

typedef enum debugOutputEnum debugOutputEnum;

struct debugOutputConfig
{
	/// @brief A bit flag of `debugOutputEnum`
	unsigned char debugOutputEnabled;
	unsigned char verbosity_debug;
	unsigned char verbosity_info;
	unsigned char verbosity_warning;
	unsigned char verbosity_error;
};

typedef struct debugOutputConfig debugOutputConfig;

// extern debugOutputConfig debugOutput;
void DebugAddSection(debugOutputConfig category, bool startOfSection, const char* format, ...);
void DebugPrintfLn(debugOutputConfig category, bool startOfSection, const char* format, ...);
#endif
